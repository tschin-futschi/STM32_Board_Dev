main() → App_Protocol_Poll() 工作原理

本文讲解从 MCU 上电到协议帧被解析执行的完整数据流。涵盖三个阶段：硬件接收（ISR + 环形缓冲区）、主循环取数据（RxRead）、协议解析（状态机 + 派发）。

二、阶段 1：硬件中断接收（ISR → 环形缓冲区）

1USART1 收到一个字节

USART1 硬件自动将串口线上的数据移入 DR（Data Register），同时将 RXNE（RX Not Empty）标志置位，触发 USART1_IRQHandler（优先级 2）。

// stm32f4xx_it.c:74-81
void USART1_IRQHandler(void)
{
    if (USART_GetITStatus(BSP_UART_PERIPH, USART_IT_RXNE) != RESET)
    {
        BSP_UART_RxISR_Callback();          // 读 DR → 自动清 RXNE
        USART_ClearITPendingBit(...);      // 冗余清除（读 DR 已清）
    }
}

2回调函数：写入环形缓冲区

回调函数在 ISR 上下文中执行，必须尽快完成。它做的事情非常简单：读取 DR 中的字节，写入环形缓冲区，推进 head 指针。

// bsp_uart.c:240-254
void BSP_UART_RxISR_Callback(void)
{
    uint8_t  byte     = (uint8_t)USART_ReceiveData(BSP_UART_PERIPH);
    uint16_t nextHead = (s_rxHead + 1U) % BSP_UART_RX_BUF_SIZE;  // 256

    if (nextHead != s_rxTail)      // 缓冲区未满
    {
        s_rxBuf[s_rxHead] = byte;  // 存入当前 head 位置
        s_rxHead = nextHead;       // head 前移
    }
    else
    {
        s_rxOverflowCount++;       // 满了就丢弃，计数
    }
}

环形缓冲区原理

s_rxBuf[256] 是一个环形数组，s_rxHead 指向下一个写入位置（ISR 推进），s_rxTail 指向下一个读取位置（主循环推进）。当 head == tail 时缓冲区为空，当 (head + 1) % 256 == tail 时为满。

   tail           head
    ↓               ↓
┌───┬───┬───┬───┬───┬───┬───┬───┐
│   │ A │ B │ C │ D │   │   │   │  ← 4 字节待读取
└───┴───┴───┴───┴───┴───┴───┴───┘
     ←── 主循环读 ──→←── ISR 写 ──→

4App_Protocol_Poll() 内部逻辑

该函数做三件事：检查异常状态、批量读取缓冲区、逐字节解析。

// app_protocol.c:905-935
void App_Protocol_Poll(void)
{
    uint16_t len;
    uint16_t i;

    // ① 检查 I2C 总线恢复事件
    if (BSP_I2C2_GetAndClearRecoveryFlag() != 0U)
    {
        App_Protocol_SendDebugInfo("I2C bus recovered");
    }

    // ② 检查 UART 接收溢出
    {
        uint32_t overflow = BSP_UART_GetAndClearOverflowCount();
        if (overflow > 0U)
        {
            // 向上位机上报丢字节数
            App_Protocol_SendDebugInfo("UART RX overflow: ...");
        }
    }

    // ③ 从环形缓冲区批量读取所有待处理字节
    len = BSP_UART_RxRead(s_rxBuf, BSP_UART_RX_BUF_SIZE);

    // ④ 逐字节喂入状态机解析
    for (i = 0U; i < len; i++)
    {
        ProcessByte(s_rxBuf[i]);
    }
}

5BSP_UART_RxRead() — 批量取出

从环形缓冲区拷贝数据到协议层的本地缓冲区。快照 head 时短暂关中断，读取过程中 ISR 可以继续写入新数据。

// bsp_uart.c:281-298
uint16_t BSP_UART_RxRead(uint8_t *pBuf, uint16_t maxLen)
{
    uint16_t count = 0U;
    uint16_t head;

    __disable_irq();
    head = s_rxHead;       // 快照当前 head（原子读取）
    __enable_irq();

    while (count < maxLen && s_rxTail != head)
    {
        pBuf[count] = s_rxBuf[s_rxTail];
        s_rxTail = (s_rxTail + 1U) % BSP_UART_RX_BUF_SIZE;
        count++;
    }
    return count;           // 返回实际读取的字节数
}

6状态机 7 个状态

收到 0xAA 收到 0x55 ┌──────────────┐ ┌──────────────┐ │ WAIT_SOF1 │───→│ WAIT_SOF2 │ └──────────────┘ └──────┬───────┘ ↑ 不匹配则 │ 匹配 │ 回到 SOF1 ▼ │ ┌──────────────┐ │ │ WAIT_SEQ │ 存 seq，初始化 CRC │ └──────┬───────┘ │ ▼ │ ┌──────────────┐ │ │ WAIT_CMD │ 存 cmd，更新 CRC │ └──────┬───────┘ │ ▼ │ ┌──────────────┐ │ │ WAIT_LEN │ 存 len，更新 CRC │ └──────┬───────┘ │ │ │ len==0? │ len>0? │ ┌─────────┴─────────┐ │ ▼ ▼ │ ┌──────────────┐ ┌──────────────┐ │ │ WAIT_CRC_H │ │ WAIT_DATA │ 逐字节存入 data[] │ └──────┬───────┘ │ 更新 CRC │ │ │ │ idx++ 直到 │ │ │ │ idx >= len │ │ │ └──────┬───────┘ │ │ │ │ │←─────────────────┘ │ ▼ │ ┌──────────────┐ │ │ WAIT_CRC_L │ 组合 CRC_H + CRC_L │ │ │ 与 s_rxCrcAccum 比较 │ └──────┬───────┘ │ │ │ 匹配？│ │ ┌─────┴──────┐ │ ▼ ▼ │ CRC OK CRC 失败 │ DispatchFrame() SendErrorResp(0xFF, ERR_CRC) │ │ │ └────┴────────────┘ 回到 WAIT_SOF1

关键代码逐段

SOF 同步：连续的 0xAA 不会打断同步，状态机会停留在 WAIT_SOF2 等待 0x55。

// app_protocol.c:819-840
case STATE_WAIT_SOF1:
    if (byte == 0xAA) { s_state = STATE_WAIT_SOF2; }
    break;

case STATE_WAIT_SOF2:
    if (byte == 0x55)      { s_state = STATE_WAIT_SEQ; }   // 帧头完成
    else if (byte == 0xAA) { s_state = STATE_WAIT_SOF2; }  // 可能是新帧开头
    else                   { s_state = STATE_WAIT_SOF1; }  // 不匹配，重来
    break;

SEQ / CMD / LEN：每收一个字段字节，都同步更新 CRC 累加器。

// app_protocol.c:842-858
case STATE_WAIT_SEQ:
    s_rxFrame.seq = byte;
    s_rxCrcAccum  = CRC16_Update(0xFFFF, byte);  // CRC 初值 0xFFFF
    s_state       = STATE_WAIT_CMD;
    break;

case STATE_WAIT_CMD:
    s_rxFrame.cmd = byte;
    s_rxCrcAccum  = CRC16_Update(s_rxCrcAccum, byte);
    s_state       = STATE_WAIT_LEN;
    break;

case STATE_WAIT_LEN:
    s_rxFrame.len = byte;
    s_rxCrcAccum  = CRC16_Update(s_rxCrcAccum, byte);
    s_dataIdx     = 0U;
    s_state       = (byte == 0U) ? STATE_WAIT_CRC_H : STATE_WAIT_DATA;
    break;                   // LEN=0 直接跳过 DATA 阶段

DATA 接收：逐字节写入 s_rxFrame.data[]，直到收满 len 个字节。

// app_protocol.c:861-868
case STATE_WAIT_DATA:
    s_rxFrame.data[s_dataIdx++] = byte;
    s_rxCrcAccum = CRC16_Update(s_rxCrcAccum, byte);
    if (s_dataIdx >= s_rxFrame.len)
    {
        s_state = STATE_WAIT_CRC_H;   // 数据收完，等 CRC
    }
    break;

CRC 校验：收到 CRC 低字节后，组合为 16-bit 值，与累加计算的值比较。

// app_protocol.c:875-883
case STATE_WAIT_CRC_L:
{
    uint16_t rxCrc = ((uint16_t)s_rxCrcHigh << 8U) | (uint16_t)byte;
    if (rxCrc != s_rxCrcAccum)
        SendErrorResp(0xFF, PROTO_ERR_CRC_FAIL);  // CRC 错误，seq 不可信用 0xFF
    else
        DispatchFrame(&s_rxFrame);               // CRC 通过，派发执行
    s_state = STATE_WAIT_SOF1;                    // 无论如何回到初始状态
    break;
}

五、阶段 4：DispatchFrame 命令派发

7根据 CMD 字段分发到对应 Handler

// app_protocol.c:735-809
static void DispatchFrame(const Proto_Frame_t *pFrame)
{
    // 批量读取进行中时，只响应心跳，忽略其他命令
    if (s_bulkReadActive && (pFrame->cmd != PROTO_CMD_HEARTBEAT))
        return;

    switch (pFrame->cmd)
    {
        case 0x00:  HandleHeartbeat(pFrame);       break;
        case 0x02:  HandleSetMotorAddr(pFrame);    break;
        case 0x03:  HandleSetBaudrate(pFrame);     break;
        case 0x04:  HandleReset(pFrame);           break;
        case 0x05:  HandleMotorPing(pFrame);       break;
        case 0x07:  HandleI2CScan(pFrame);         break;
        case 0x08:  HandlePmicEnable(pFrame);      break;
        case 0x09:  HandlePmicSetVolt(pFrame);     break;
        case 0x0A:  HandlePmicDisable(pFrame);     break;
        case 0x20:  HandleReadReg(pFrame);         break;
        case 0x21:  HandleWriteReg(pFrame);        break;
        case 0x22:  HandleBulkRead(pFrame);        break;
        case 0x50:  HandleStartSample(pFrame);     break;
        case 0x51:  HandleStopSample(pFrame);      break;
        case 0x52:  HandleSetInterval(pFrame);     break;
        case 0x53:  HandleSetChannel(pFrame);      break;
        case 0x54:  HandleSetRegMap(pFrame);       break;
        case 0x55:  HandleStartLinearGen(pFrame);  break;
        case 0x56:  HandleStartCosineGen(pFrame); break;
        case 0x57:  HandleStopGenerator(pFrame);   break;
        default:   SendErrorResp(pFrame->seq, PROTO_ERR_UNKNOWN_CMD); break;
    }
}
命令码范围分组典型命令
0x00~0x1F系统控制心跳、设地址、改波特率、复位、I2C 扫描、PMIC 控制
0x20~0x4F寄存器读写单读、单写、批量读
0x50~0x7F示波器/发生器启停采样、设频率、设通道、设寄存器映射、线性/余弦发生器

六、响应发送路径

8SendFrame() 构建响应帧 + DMA 发送

每个 Handler 执行完业务逻辑后，调用 SendFrame() 回复上位机。

// app_protocol.c:119-147
static ErrorStatus SendFrame(uint8_t seq, uint8_t cmd,
                              const uint8_t *pData, uint8_t len)
{
    static uint8_t s_txBuf[PROTO_MAX_FRAME_LEN];  // 262 字节
    uint16_t crc = 0xFFFF;
    uint16_t idx = 0U;

    s_txBuf[idx++] = 0xAA;                // SOF1
    s_txBuf[idx++] = 0x55;                // SOF2
    s_txBuf[idx] = seq; crc = CRC16_Update(crc, seq); idx++;
    s_txBuf[idx] = cmd; crc = CRC16_Update(crc, cmd); idx++;
    s_txBuf[idx] = len; crc = CRC16_Update(crc, len); idx++;

    for (i = 0U; i < len; i++) {          // 填充数据 + 更新 CRC
        s_txBuf[idx] = pData[i];
        crc = CRC16_Update(crc, pData[i]);
        idx++;
    }

    s_txBuf[idx++] = (uint8_t)(crc >> 8U); // CRC 高字节
    s_txBuf[idx++] = (uint8_t)(crc);       // CRC 低字节

    BSP_UART_TxWait();                     // 等上一帧发完
    return BSP_UART_Transmit(s_txBuf, idx); // DMA 启动发送
}

发送不阻塞主循环

BSP_UART_Transmit() 将数据交给 DMA2 Stream7 Channel4，DMA 硬件自动逐字节搬运到 USART1 TX。CPU 立即返回继续执行主循环。DMA 传输完成后触发 DMA2_Stream7_IRQHandler，将 s_txDone = 1。

七、一次完整交互的时序

以上位机发送一个心跳帧为例：AA 55 01 00 00 40 04

时间轴 → PC 发送: AA 55 01 00 00 40 04 │ │ │ │ │ │ │ ISR 逐字节 ▼ ▼ ▼ ▼ ▼ ▼ ▼ 写入环形 [AA][55][01][00][00][40][04] 缓冲区 ↑ ↑ s_rxTail s_rxHead ─── 下一次主循环 ─── App_Protocol_Poll(): ├─ BSP_UART_RxRead() → 读出 7 字节 ├─ ProcessByte(0xAA) → state: SOF1 → SOF2 ├─ ProcessByte(0x55) → state: SOF2 → SEQ ├─ ProcessByte(0x01) → seq=1, CRC 初始化, state → CMD ├─ ProcessByte(0x00) → cmd=0x00, state → LEN ├─ ProcessByte(0x00) → len=0, state → CRC_H (跳过 DATA) ├─ ProcessByte(0x40) → crcHigh=0x40, state → CRC_L └─ ProcessByte(0x04) → rxCrc=0x4004, 校验通过 └─ DispatchFrame(cmd=0x00) └─ HandleHeartbeat() └─ SendFrame(seq=1, cmd=0x00, NULL, 0) └─ DMA 发送: AA 55 01 00 00 40 04 → PC 收到心跳回复

八、关键设计要点总结

设计点实现方式作用
接收解耦ISR 写环形缓冲区，主循环读ISR 快进快出（~20 条指令），不丢字节
帧同步双字节帧头 0xAA 0x55在字节流中定位帧边界，容忍噪声
完整性校验CRC16 逐字节累加检测传输错误，CRC 失败用固定 seq=0xFF 回复
发送非阻塞DMA 搬运 + TC 中断CPU 不等待串口发完，继续处理下一个任务
状态机解析每字节切换一次状态天然支持拆包（一帧可能跨多次 Poll 到达）
错误隔离CRC 错误不影响状态机报错后立即回到 SOF1，等待下一帧
