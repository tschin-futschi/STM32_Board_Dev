AW86008/AW86100 本地 ISP 烧录流程分析

日期： 2026-05-22 ｜ 对应路线图： 4.6 (待调试) ｜ 命令码： 0x32 ~ 0x37

顶层视角。 PC 把固件分块经 UART 灌进 STM32 SRAM1 staging buffer，攒齐后 STM32 阻塞调用一次 AW_86008_86100_Flash_Run()，把固件通过 I2C2（软件 bit-banging）烧进 AW 芯片，再让芯片跳出 UBOOT 运行新固件。整个 EXEC 阶段主循环停摆 5-10s。

5 层抽象

层文件职责状态
① 协议层 (PC ↔ STM32)App/Src/app_protocol.c状态机 + 64 KB staging完成
② 顶层封装 (3 步流程)Flash/AW/AW86008_86100/Src/aw_flash.cboot → download → reset完成
③ ISP 协议层 (vendor)Flash/AW/AW86008_86100/Src/aw_uboot_isp.cUBOOT 帧协议 + Flash 操作jump_check 是 stub

⑤ I2C2 BSP (软件 I2C)BSP/Src/bsp_i2c2.cbit-banging（HW 达不到 1 MHz）完成

每条命令

CMD名data 段resp关键校验
0x32FLASH_BEGIN[addr(4 LE)][totalBytes(4 LE)]空 addr ∈ [0x01000000, 0x01020000)；totalBytes 4 字节对齐且 ≤ 64 KB；addr+totalBytes 不越界
0x33FLASH_DATA[pktSeq(2 LE)][chunk(N)][nextSeq(2 LE)] state==RECEIVING；pktSeq 与 s_fwNextSeq 严格相等（不允许乱序）；累计不溢出

0x36FLASH_CANCEL空空无条件 → IDLE
0x37FLASH_RESET_CHIP空[ispStatus(1)]独立调 aw_reset_chip，不动 session

HandleFlashExec 关键代码

/* app_protocol.c:1051 */
static void HandleFlashExec(const Proto_Frame_t *pFrame)
{
    ISP_STATUS_E ret;
    uint8_t resp[1];

    /* 入口互斥 1：状态必须是 READY */
    if (s_fwState != (uint8_t)FLASH_STATE_READY) {
        SendErrorResp(pFrame->seq, PROTO_ERR_EXEC_FAIL); return;
    }

    /* 入口互斥 2：I2C2 共用资源 — 采样 / 生成器 / 批量读都不能在跑 */
    if ((App_Sample_IsActive() != 0U) ||
        (App_Generator_IsRunning() != 0U) ||
        (s_bulkReadActive != 0U)) {
        SendErrorResp(pFrame->seq, PROTO_ERR_EXEC_FAIL); return;
    }

    /* 主动告知 PC：5-10s 阻塞窗口开启，请扩大心跳超时 */
    App_Protocol_SendDebugInfo("FLASH_EXEC START");
    BSP_UART_TxWait();

    s_fwState = (uint8_t)FLASH_STATE_EXECUTING;
    ret = AW_86008_86100_Flash_Run(s_fwAddr, s_fwBuf, s_fwTotalBytes / 4U);   /* <<< 阻塞 5-10s */
    s_fwState = (ret == ISP_OK) ? (uint8_t)FLASH_STATE_DONE : (uint8_t)FLASH_STATE_ERROR;
    s_flashSessionActive = 0U;

    resp[0] = (uint8_t)ret;
    (void)SendFrameWithRetry(pFrame->seq, (uint8_t)PROTO_CMD_FLASH_EXEC, resp, 1U);
}
注意。 s_fwBuf[64KB] 静态分配在 SRAM1，已经占了 SRAM1 总容量 112 KB 的 57%。后续若新增大对象需精打细算。

Layer 2 ── 顶层封装（aw_flash.c）

/* aw_flash.c:8 — 一站式入口 */
ISP_STATUS_E AW_86008_86100_Flash_Run(uint32_t addr, uint8_t *bin_buf, uint32_t len)
{
    ISP_STATUS_E ret;

    /* Step 1: 进 UBOOT + stay-in-uboot */
    ret = aw_boot_control();
    if (ret != ISP_OK) return ret;

    /* Step 2: 擦 + 写 + jump_check (stub) */
    ret = aw_i2c_isp_download(addr, bin_buf, len);
    if (ret != ISP_OK) return ret;

    /* Step 3: wake-out-of-uboot 跑新固件 */
    ret = aw_reset_chip();
    if (ret != ISP_OK) return ret;

    return ISP_OK;
}

注意：len 单位是 字（4 字节），不是字节。HandleFlashExec 传参时做了 s_fwTotalBytes / 4U 的转换。

Layer 3 ── ISP 协议层（vendor，aw_uboot_isp.c）

3.1 共享子函数：AwWakeSequence

boot_control 和 reset_chip 共用此 7 写 + 1 读校验序列：

/* aw_uboot_isp.c:108 */
static ISP_STATUS_E AwWakeSequence(void)
{
    /* 7 步级 I2C 写 (DevAddr=0x6B = AW_CHIP_I2C_ADDR) */
    aw_i2c_write(0x6B, 0, NULL, 3, {0xff, 0xa3, 0x02}); delay_ms(2);
    aw_i2c_write(0x6B, 0, NULL, 3, {0xff, 0xb7, 0x5a}); delay_ms(2);
    aw_i2c_write(0x6B, 0, NULL, 3, {0xff, 0xb7, 0x00}); delay_ms(15);
    aw_i2c_write(0x6B, 0, NULL, 10,{0xff,0xf0,0x20,0x20,0x02,0x02,0x19,0x29,0x19,0x29});

    /* 1 步读：[0xff,0xf0] 期望 0x01 — 整个序列的唯一硬校验 */
    uint8_t rdval = 0;
    aw_i2c_read(0x6B, 2, {0xff,0xf0}, 1, &rdval);
    if (rdval != 0x01) return ISP_FLASH_ERROR;   /* <<< 校验失败的唯一入口 */
    delay_ms(2);

    aw_i2c_write(0x6B, 0, NULL, 3, {0xff, 0xb4, 0x00}); delay_ms(2);
    aw_i2c_write(0x6B, 0, NULL, 3, {0xff, 0xa3, 0x00}); delay_ms(2);
    aw_i2c_write(0x6B, 0, NULL, 3, {0xff, 0xb4, 0x01});

    return ISP_OK;
}

3.2 aw_boot_control = AwWakeSequence + 20× stay-in-uboot

ISP_STATUS_E aw_boot_control(void)
{
    ISP_STATUS_E ret = AwWakeSequence();
    if (ret != ISP_OK) return ret;
    delay_ms(7);

    /* stay-in-uboot：写 0xAC 到 0x6F (=AW_UBOOT_I2C_ADDR)，连续 20 次，每次 1ms */
    /* 必须在芯片上电后 8ms 窗口内完成，否则会跑出 UBOOT 进入用户固件 */
    for (uint8_t i = 0; i < 20; i++) {
        aw_i2c_stop_uboot();    /* 一字节 0xAC 写到 0x6F */
        delay_ms(1);
    }
    return ISP_OK;
}

3.3 aw_i2c_isp_download — 实际烧录

ISP_STATUS_E aw_i2c_isp_download(uint32_t addr, uint8_t *bin_buf, uint32_t len)
{
    /* 4 步前置校验：ops 注册、bin_buf 非空、addr 在区间、Flash 空间够 */
    if (s_ops == NULL)                     return ISP_NOT_INITED;
    if (bin_buf == 0)                     return ISP_PBUF_ERROR;
    if (addr < 0x01000000 || addr >= 0x01020000) return ISP_ADDR_ERROR;
    if ((0x01020000 - addr) < len*4)   return ISP_SPACE_ERROR;

    /* 3 步真烧录： */
    if (aw_hank_connect_check()             != ISP_OK) return ISP_HANK_ERROR;   /* 握手帧到 0x6F */
    if (aw_flash_download_check(addr,bin_buf,len) != ISP_OK) return ISP_FLASH_ERROR;  /* 擦+写 */
    if (aw_flash_jump_check(addr)           != ISP_OK) return ISP_JUMP_ERROR;   /* <<< stub! */

    return ISP_OK;
}

3.4 aw_flash_download_check — 擦 + 写 拆块策略

/* aw_uboot_isp.c:474 */
ISP_STATUS_E aw_flash_download_check(uint32_t addr, uint8_t *bin_buf, uint32_t len)
{
    /* 拆块（单位：字）：每 16 字 = 64 字节写一块；每 1024 字 = 4 KB 擦一块 */
    uint32_t divisor   = len / 16;            /* 完整 16-字 写块数 */
    uint8_t  remaind   = len % 16;            /* 尾部不足 16 字的字数 */
    uint32_t block_cnt = len / 1024 + ((len % 1024) ? 1 : 0);  /* 1024-字擦块数 */

    /* Step A: 擦除（一次性按 1024 字 = 4 KB 块擦完所有要用的区域） */
    if (aw_flash_block_erase_check(addr, block_cnt) != ISP_OK) return ISP_FLASH_ERROR;

    /* Step B: 整 16-字块循环写入 */
    for (i = 0; i < divisor; i++) {
        if (aw_flash_block_write_ckeck(addr, i, bin_buf, 16) != ISP_OK) return ISP_FLASH_ERROR;
    }

    /* Step C: 写不足 16 字的尾巴（如果有） */
    if (remaind != 0) {
        if (aw_flash_block_write_ckeck(addr, i, bin_buf, remaind) != ISP_OK) return ISP_FLASH_ERROR;
    }
    return ISP_OK;
}
子操作UBOOT 帧 event写帧长读 ACK 长延时
hank_connectHANK_CONNECT (0x01)9 B14 B2 ms
block_erase (1 块)FLASH_ERASE_BLOCK (0x21)15 B10 B25 × len ms（按擦块数线性放大）
block_write (1 块)FLASH_WRITE (0x01)13+4·len B10 B1 ms
jump_check—stub，无 I2C 操作，直接 return ISP_OK
关键风险 R1 ── jump_check 是 stub。
源码 aw_uboot_isp.c:249-298 注释明确写：
"此函数永远返回 ISP_OK。jump 后是否真正运行新固件无法在此层判断。调用此函数后 aw_i2c_isp_download 返回 ISP_OK 仅等于 'erase + write 子步骤都 ACK 了'，不等于新固件已运行。PC 端必须用'烧录后回读 Flash 校验'或 '0x37 RESET_CHIP + 后续操作可达性测试' 二次确认烧录结果。"
联调阶段如发现"EXEC 返 OK 但芯片没跑新固件"，第一嫌疑就是这里。

3.5 aw_reset_chip ── 实为 wake-out-of-uboot

ISP_STATUS_E aw_reset_chip(void)
{
    /* 复用 boot_control 的 wake 序列，但 *不* 接 stay-in-uboot 循环 */
    /* 等价于"让芯片从 Flash 0x01000000 跑用户固件" — 不是硬件 reset，沿用 vendor 命名 */
    ISP_STATUS_E ret = AwWakeSequence();
    if (ret != ISP_OK) return ret;
    delay_ms(15);
    return ISP_OK;
}

Layer 4 ── Port 适配（aw_port_stm32.c）

通过 aw_isp_ops_t 回调表把 vendor 调用翻译到 BSP，由 main.c 在初始化时 aw_isp_init(&g_awOpsStm32) 注入：

const aw_isp_ops_t g_awOpsStm32 = {
    .delay_ms  = AwPort_DelayMs,    /* → BSP_GetTick() 忙等 */
    .i2c_write = AwPort_I2cWrite,   /* → BSP_I2C2_TransparentWrite */
    .i2c_read  = AwPort_I2cRead,    /* → BSP_I2C2_TransparentRead  */
};

Signature 翻译细节：

  • 参数顺序：AW (DevId, AddrSize, pAddr, Size, Data) ↔ BSP (devAddr, pAddr, addrSize, pData, dataLen)
  • 返回类型：AW int (0=OK) ↔ BSP ErrorStatus (SUCCESS/ERROR)
  • 长度：AW uint8_t ↔ BSP uint16_t（隐式提升，无损）

Layer 5 ── I2C2 软件 bit-banging

I2C2（PB10/PB11）软件实现而非硬件外设，因为 AW UBOOT 协议需要 ~1 MHz 时钟，而 STM32F4 硬件 I2C 上限 400 kHz。所有底层字节收发走 BSP_I2C2_TransparentWrite/Read 的 GPIO 模拟。
