2. BSP 层 — 软件位拼透传

2.1 BSP_I2C2_TransparentWrite

bsp_i2c2.c

ErrorStatus BSP_I2C2_TransparentWrite(uint8_t devAddr,
                                       const uint8_t *pAddr, uint8_t addrSize,
                                       const uint8_t *pData, uint16_t dataLen)
{
    uint16_t i;

    I2C_Start();
    if (!SendByte((uint8_t)(devAddr << 1U)))             goto error;  // addr+W

    for (i = 0U; i < addrSize; i++)                                   // ① 可选地址段
    {
        if (!SendByte(pAddr[i]))                          goto error;
    }

    for (i = 0U; i < dataLen; i++)                                    // ② 数据段
    {
        if (!SendByte(pData[i]))                          goto error;
    }

    I2C_Stop();
    return SUCCESS;

error:
    I2C_Stop();
    BSP_I2C2_RecoverBus();                                           // 9 SCL 脉冲 + SoftReset
    return ERROR;
}

2.2 BSP_I2C2_TransparentRead

ErrorStatus BSP_I2C2_TransparentRead(uint8_t devAddr,
                                      const uint8_t *pAddr, uint8_t addrSize,
                                      uint8_t *pData, uint16_t dataLen)
{
    uint16_t i;

    if (dataLen == 0U) { return SUCCESS; }                            // 空读直接 OK

    if (addrSize > 0U)                                                // ── 路径 A: 标准寄存器读
    {
        I2C_Start();
        if (!SendByte((uint8_t)(devAddr << 1U)))         goto error;  // addr+W
        for (i = 0U; i < addrSize; i++)
        {
            if (!SendByte(pAddr[i]))                      goto error; // 写地址段
        }

        I2C_RepeatedStart();                                          // ★ 关键: 切换方向
        if (!SendByte((uint8_t)((devAddr << 1U) | 0x01U))) goto error;// addr+R
    }
    else                                                              // ── 路径 B: 裸读
    {
        I2C_Start();
        if (!SendByte((uint8_t)((devAddr << 1U) | 0x01U))) goto error;// 直接 addr+R
    }

    for (i = 0U; i < dataLen; i++)
    {
        pData[i] = RecvByte((i < (dataLen - 1U)) ? 1U : 0U);          // 末位 NACK
    }

    I2C_Stop();
    return SUCCESS;

error:
    I2C_Stop();
    BSP_I2C2_RecoverBus();
    return ERROR;
}

2.3 BSP 层 3 个要点

  1. addrSize == 0 直接跳过 ① 循环，无需特判分支
  2. 读路径分两条：addrSize > 0 走 START + addr+W + AddrBytes + RepeatedStart + addr+R；addrSize == 0 只有一次 START + addr+R
  3. 读字节循环 (i < dataLen-1) ? 1 : 0 决定主机 ACK / NACK：前 N-1 字节 ACK 让从机继续吐，最后一字节 NACK 告知读完

3. 协议层 — Handler 实现

3.1 HandleI2cPassWrite (0x30) — 5 步校验

app_protocol.c

static void HandleI2cPassWrite(const Proto_Frame_t *pFrame)
{
    uint8_t        devAddr;
    uint8_t        addrSize;
    uint8_t        dataLen;
    const uint8_t *pAddr;
    const uint8_t *pData;
    uint16_t       expectedLen;

    // ① 载荷下限: DevId(1) + AddrSize(1) + DataLen(1) = 3 字节
    if (pFrame->len < 3U)
    {
        SendErrorResp(pFrame->seq, PROTO_ERR_EXEC_FAIL);
        return;
    }

    devAddr  = pFrame->data[0];
    addrSize = pFrame->data[1];

    // ② DevId 7-bit 范围 (0x00 ~ 0x7F)
    if (devAddr > PROTO_MOTOR_ADDR_MAX)
    {
        SendErrorResp(pFrame->seq, PROTO_ERR_EXEC_FAIL);
        return;
    }

    // ③ 头 3 字节 + AddrBytes 段必须能装下
    if ((uint16_t)pFrame->len < (uint16_t)(3U + addrSize))
    {
        SendErrorResp(pFrame->seq, PROTO_ERR_EXEC_FAIL);
        return;
    }

    dataLen     = pFrame->data[2U + addrSize];                        // DataLen 字段位置浮动
    expectedLen = (uint16_t)(3U + addrSize) + (uint16_t)dataLen;

    // ④ LEN 与 (3 + AddrSize + DataLen) 完整一致性
    if ((uint16_t)pFrame->len != expectedLen)
    {
        SendErrorResp(pFrame->seq, PROTO_ERR_EXEC_FAIL);
        return;
    }

    // ⑤ 透传到 BSP, pAddr / pData 在长度=0 时传 NULL
    pAddr = (addrSize > 0U) ? &pFrame->data[2]                : NULL;
    pData = (dataLen  > 0U) ? &pFrame->data[3U + addrSize]    : NULL;

    if (BSP_I2C2_TransparentWrite(devAddr, pAddr, addrSize, pData, dataLen) != SUCCESS)
    {
        SendErrorResp(pFrame->seq, PROTO_ERR_EXEC_FAIL);
        return;
    }

    (void)SendFrame(pFrame->seq, (uint8_t)PROTO_CMD_I2C_PASS_WRITE, NULL, 0U);
}
为什么校验分两段（③、④）？ 先用 ③ 确认能安全读到 DataLen 字段（避免越界访问 pFrame->data），再用 DataLen 算总长一致性。 (uint16_t) 强制提升防溢出：uint8_t + uint8_t + uint8_t 理论值最大 765，提升后 uint16_t 不会回绕。

3.2 HandleI2cPassRead (0x31)

static void HandleI2cPassRead(const Proto_Frame_t *pFrame)
{
    static uint8_t s_passReadBuf[255U];        // 响应缓冲, LEN 字段上限 255

    uint8_t        devAddr;
    uint8_t        addrSize;
    uint8_t        readLen;
    const uint8_t *pAddr;

    if (pFrame->len < 3U)                       → err;
    devAddr  = pFrame->data[0];
    addrSize = pFrame->data[1];
    if (devAddr > PROTO_MOTOR_ADDR_MAX)         → err;

    // ★ 严格相等: 读命令没有 Data 段
    if ((uint16_t)pFrame->len != (uint16_t)(3U + addrSize))
    {
        SendErrorResp(pFrame->seq, PROTO_ERR_EXEC_FAIL);
        return;
    }

    readLen = pFrame->data[2U + addrSize];
    if (readLen == 0U)                          → err;   // 主动拒绝空读

    pAddr = (addrSize > 0U) ? &pFrame->data[2] : NULL;

    if (BSP_I2C2_TransparentRead(devAddr, pAddr, addrSize,
                                  s_passReadBuf, readLen) != SUCCESS) → err;

    // ★ 响应数据段 = 读到的 readLen 字节
    (void)SendFrame(pFrame->seq, (uint8_t)PROTO_CMD_I2C_PASS_READ,
                    s_passReadBuf, readLen);
}

3.3 与 0x30 的两点关键差异

  • LEN 校验是严格相等 len == 3 + addrSize（读命令载荷里只有 ReadLen 而无 Data）
  • 响应数据段非空：readLen 字节读到的内容直接装在响应载荷里
  • 主动拒绝 readLen == 0（无意义且会让 BSP 早退）

switch (pFrame->cmd) {

4. 关键设计决策

决策理由
BSP 层不解析 Data 段语义 协议明文要求"原样透传，不得解析或重组"
s_passReadBuf 用文件级 static 而非栈 255 B 在主循环栈上还放得下，但 SendFrame 还要一份 PROTO_MAX_FRAME_LEN 的 s_txBuf；用 static 把缓冲固化到 BSS 更可控（bss +256 B）
AddrSize 不做上限校验 按协议放行（v2.5 文档未限定上限）；pFrame->len < 3 + addrSize 自动拦截病态值（uint8_t LEN 最大 255 限制了实际范围）
失败统一回 0x03 不区分子类型（参数错 / I2C NACK / 总线超时），符合协议 v1.4+ 错误码定义
BSP 失败不本地重试 CLAUDE.md 第 6 节："BSP 返回 ERROR 时，应用层直接 NACK，不做本地重试，重试策略由上位机决定"
校验顺序固定：长度 → DevId → 浮动字段位置 → 总长 → 透传 越靠前的校验越廉价，避免对越界字节的访问；强制 (uint16_t) 提升防止 uint8_t 算术回绕

5. 错误路径汇总

触发条件响应字节错误码
CRC16 校验失败（状态机层拦截） AA 55 FF 01 01 01 CRCH CRCL 0x01 (SEQ=0xFF)
pFrame->len < 3 AA 55 SEQ 01 01 03 ... 0x03
DevId > 0x7F 同上0x03
